--ERD (Entity-Relationship Diagram) : 개체와 관계로 시나리오를 도식화
	>>ERD는 개념적 설계과정
		>논리적 설계 (테이블 명세서)>물리적 설계 (쿼리문)
	
-E: 개체: 독립적으로 존재할 수 있는 대상 (개체!=테이블)
	개체>직사각형으로 표현// 속성> 원으로 표현// 기본키>밑줄로 표현
-R: 관계
	관계>마름모로 표현
	1:1 관계
	1:N 관계		} >각 개체관점에서 어떤 관계인지 확인
	N:M 관계

	ex)교수는 id, 이름
	교수는 논문을 작성
	논문은 작성일, 아이디만 저장
	>> 교수(E)~"작성(R)"~ 논문(E)
	
	ex)고객이 있고 고객은 출석체크
	출석시간을 저장
	1명은 여러번 출석
	>>고객개체와 출석시간 개체는 출석관계가 있고, 1:N 관계를 가진다.
		고객은 무조건 출석해야한다는 강제성을 가진다.
		고객은 출석하지 않아도 된다.

--쿼리문 작성
create table member(
id varchar2(8) primary key,
name varchar2(10),
gender varchar2(2) check (gender in ('남','여')),
point number(6,2)
);

create table event(
evtnum number(3) primary key,
evtname varchar2(20),
evtdate timestamp
);

create table participation(
ptcnum number(3) primary key,
evtnum number(3),
id varchar2(8),
strory varchar2(100),
constraint fk_evtnum foreign key(evtnum) references event(evtnum)
on delete set null,		****
constraint fk_id foreign key(id) references member(id)
on delete set cascade	****
);


--외래키 제약조건
on delete set null // 부모 테이블의 컬럼이 삭제되면 자식 릴레이션의 외래키 값을 null로 설정한다.
on delete set cascade //부모 테이블의 컬럼이 삭제되면 자식 튜플도 삭제한다.
